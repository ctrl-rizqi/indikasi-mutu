
=========================== frontend =========================
# Build stage
FROM node:25-alpine AS builder
WORKDIR /app

# Install pnpm
RUN npm install -g pnpm

# Copy package files
COPY package.json pnpm-lock.yaml* pnpm-workspace.yaml ./
COPY apps/frontend/package.json ./apps/frontend/
COPY packages/shared/package.json ./packages/shared/

# Install dependencies
RUN pnpm install

# Copy source code
COPY . .

# Build shared package first
RUN pnpm run --filter @antrol/shared build

# Build frontend
RUN pnpm run --filter frontend build

# Production stage
FROM nginx:alpine AS production
COPY --from=builder /app/apps/frontend/dist /usr/share/nginx/html
COPY apps/frontend/nginx.conf /etc/nginx/conf.d/default.conf
EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]

=========================== backend =========================
# Build stage
FROM node:25-alpine AS builder
WORKDIR /app

# Install pnpm
RUN npm install -g pnpm

# Copy package files
COPY package.json pnpm-lock.yaml* pnpm-workspace.yaml ./
COPY apps/service/package.json ./apps/service/
COPY packages/shared/package.json ./packages/shared/

# Install dependencies
RUN pnpm install

# Copy source code
COPY . .

# Build backend
RUN cd apps/service && pnpm prisma generate
RUN pnpm run --filter service build

# Prepare production dependencies (isolates workspace packages & removes symlinks)
RUN rm -rf pruned
RUN pnpm --filter service --prod deploy --legacy pruned

# Production stage
FROM node:25-alpine AS production
WORKDIR /app

# Install pnpm
RUN npm install -g pnpm

COPY --from=builder /app/pruned ./

# Copy built application & prisma schema
COPY --from=builder /app/apps/service/dist ./dist
COPY --from=builder /app/apps/service/prisma ./prisma

# Generate Prisma Client in production to ensure binary compatibility
RUN pnpm add prisma@7.3.0
RUN pnpm exec prisma generate

EXPOSE 3000
CMD ["node", "dist/main.js"]

====================== docker-compose.yml ====================
services:
  mysql:
    image: mysql:8.0
    container_name: antrol-mysql
    restart: unless-stopped
    environment:
      MYSQL_ROOT_PASSWORD: root123
      MYSQL_DATABASE: antrol_service
      MYSQL_USER: antrol
      MYSQL_PASSWORD: antrol123
    ports:
      - "3306:3306"
    volumes:
      - mysql_data:/var/lib/mysql
      - ./docker/mysql/init:/docker-entrypoint-initdb.d
    command: --default-authentication-plugin=mysql_native_password --character-set-server=utf8mb4 --collation-server=utf8mb4_unicode_ci
    healthcheck:
      test: ["CMD", "mysqladmin", "ping", "-h", "localhost", "-u", "root", "-proot123"]
      interval: 10s
      timeout: 5s
      retries: 5
    networks:
      - antrol-network

  backend:
    build:
      context: .
      dockerfile: apps/service/Dockerfile.backend
      target: production
    container_name: antrol-backend
    restart: unless-stopped
    env_file:
      - .env.docker
    ports:
      - "3000:3000"
    depends_on:
      mysql:
        condition: service_healthy
    volumes:
      - /etc/localtime:/etc/localtime:ro  # sync dengan host
      - /etc/timezone:/etc/timezone:ro
      - ./apps/service/prisma:/app/prisma
    networks:
      - antrol-network

  frontend:
    build:
      context: .
      dockerfile: apps/frontend/Dockerfile.frontend
      target: production
    container_name: antrol-frontend
    restart: unless-stopped
    ports:
      - "3001:80"
    depends_on:
      - backend
    networks:
      - antrol-network

networks:
  antrol-network:
    driver: bridge

volumes:
  mysql_data:
    driver: local
